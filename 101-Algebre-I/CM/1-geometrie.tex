\dfn{produit scalaire}{
	Soient $u = \begin{pmatrix}u_1 \\ u_2 \\ \vdots \\ u_n \end{pmatrix}, v = \begin{pmatrix}v_1 \\ v_2 \\ \vdots \\ v_n \end{pmatrix}$
	deux vecteurs de $\R^n$.
	Le \emphindex{produit scalaire} entre $u$ et $v$ est donné par
		\[ \sum_{i=1}^n u_iv_i =  u_1 v_1 + u_2v_2 + \dots + u_nv_n. \]
}{dfn:produit-scalaire}

\notation{
	Le produit scalaire entre $u$ et $v$ est lu « $u$ scalaire $v$ » et admet plusieurs notations :
	%\begin{multicols}{2}
	\begin{enumerate}
		\item
		$u \cdot v$, d'où le \emph{dot product} ;
		\item
		$\scal{u, v}$, utilisée pour les produits bilinéaires ;
		\item
		$u^T v$, produit matriciel ;
		\item
		$u_iv_i$, convention de sommation d'Einstein.
	\end{enumerate}
	%\end{multicols}
}

\nt{
	Le produit scalaire d'un vecteur $u$ avec lui-même égale le carré de sa norme.
		\[ \scal{u, u} = \norm{u}^2. \]
}

\thm{bilinéarité du produit scalaire}{
	Le produit scalaire $\scal{. , .}$ est un \emph{produit bilinéaire}.
}{thm:bilinearite-scalaire}

\dfn{projection orthogonale}{
	$\proj_u(v)$.
}{dfn:proj-orth}

\thm{}{
	Soient $u$ et $v$ deux vecteurs non nuls de $\R^n$.
	Alors $\proj_u(v)$ est le vecteur le plus proche de $v$ appartenant à $\scal{u}$.
}{thm:orth-proj-closest}

\thm{}{
	Soient $u$ et $v$ deux vecteurs non nuls de $\R^n$.
	Alors 	
		\[ \proj_u(v) = \frac{\scal{u, v}}{\scal{v, v}} v. \]
	L'application $\proj_u$ est donc linéaire.
}{thm:proj-orth-formula}
